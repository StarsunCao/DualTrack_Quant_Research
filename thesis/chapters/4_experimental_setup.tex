\section{Evaluation Metrics}

\subsection{Financial Metrics}

\subsubsection{Sharpe Ratio}

\begin{equation}
\text{Sharpe} = \frac{E[R_p - R_f]}{\sigma_p}
\end{equation}

\subsubsection{Maximum Drawdown}

\begin{equation}
\text{MDD} = \max_{t} \frac{P_{peak}(t) - P(t)}{P_{peak}(t)}
\end{equation}

\subsection{Engineering Metrics}

\begin{itemize}
    \item \textbf{Latency}: Time to generate a single signal
    \item \textbf{Throughput}: Signals per second
    \item \textbf{Cost-per-Alpha}: API cost per unit of alpha
\end{itemize}

\subsection{Advanced Metrics}

\begin{itemize}
    \item \textbf{MAE/MFE}: Adverse/Excursion analysis
    \item \textbf{Win Rate}: Percentage of profitable trades
    \item \textbf{Profit Factor}: Gross profit / Gross loss
\end{itemize}
